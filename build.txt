  	// Non of this will work without major changes to RELIC as it cannot take additional
  	// parameters today that would allow the directory name to be identified etc. Plus it requires 
  	// the ability to get the content and build the list accordingly..
  	// RPG programs
  	CRTSRCPF FILE(&INTOLIB/CBX999SRC) RCDLEN(150)
  	ADDPFM FILE(&INTOLIB/CBX999SRC) MBR(CBX999) SRCTYPE(RPGLE)
  	CPYFRMSTMF FROMSTMF('&DIR/cbx999/cbx999.rpgle') TOMBR('/QSYS.LIB/&INTOLIB.LIB/CBX999SRC.FILE/CBX999.MBR')
  	CRTRPGMOD MODULE(&INTOLIB/CBX999)  DBGVIEW(*LIST) 
	CRTPGM PGM(&INTOLIB/CBX999) ACTGRP(*NEW)
  	ADDPFM FILE(&INTOLIB/CBX999SRC) MBR(CBX999E) SRCTYPE(RPGLE)
  	CPYFRMSTMF FROMSTMF('&DIR/cbx999/cbx999e.rpgle') TOMBR('/QSYS.LIB/&INTOLIB.LIB/CBX999SRC.FILE/CBX999E.MBR')	
	CRTRPGMOD MODULE(&INTOLIB/CBX999E)  DBGVIEW(*LIST) 
	CRTPGM PGM(&INTOLIB/CBX999E) ACTGRP(*NEW)
  	ADDPFM FILE(&INTOLIB/CBX999SRC) MBR(CBX999V) SRCTYPE(RPGLE)
  	CPYFRMSTMF FROMSTMF('&DIR/cbx999/cbx999v.rpgle') TOMBR('/QSYS.LIB/&INTOLIB.LIB/CBX999SRC.FILE/CBX999V.MBR')	
	CRTRPGMOD MODULE(&INTOLIB/CBX999V)  DBGVIEW(*LIST) 
	CRTPGM PGM(&INTOLIB/CBX999V) ACTGRP(*NEW)
	// command
  	ADDPFM FILE(&INTOLIB/CBX999SRC) MBR(WRKRMTOUTQ) SRCTYPE(RPGLE)
  	CPYFRMSTMF FROMSTMF('&DIR/cbx999/wrkrmtoutq.cmd') TOMBR('/QSYS.LIB/&INTOLIB.LIB/CBX999SRC.FILE/WRKRMTOUTQ.MBR')	
	CRTCMD CMD(&INTOLIB/WRKRMTOUTQ) PGM(*LIBL/CBX999) SRCFILE(&INTOLOB/CBX999SRC)
	// panel group
	ADDPFM FILE(&INTOLIB/CBX999SRC) MBR(CBX999H) SRCTYPE(PNLGRP)
  	CPYFRMSTMF FROMSTMF('&DIR/cbx999/cbx999h.pnlgrp') TOMBR('/QSYS.LIB/&INTOLIB.LIB/CBX999SRC.FILE/CBX999H.MBR')
	ADDPFM FILE(&INTOLIB/CBX999SRC) MBR(CBX999P) SRCTYPE(PNLGRP)
  	CPYFRMSTMF FROMSTMF('&DIR/cbx999/cbx999p.pnlgrp') TOMBR('/QSYS.LIB/&INTOLIB.LIB/CBX999SRC.FILE/CBX999P.MBR')
  	CRTPNLGRP PNLGRP(&INTOLIB/CBX999H) SRCFILE(&INTOLIB/CBX999SRC)
  	CRTPNLGRP PNLGRP(&INTOLIB/CBX999P) SRCFILE(&INTOLIB/CBX999SRC)
  	// clean up
  	DLTF FILE(&INTOLIB/CBX999SRC)
